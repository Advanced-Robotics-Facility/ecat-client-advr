121inline void EcThread::create ( int rt=true, int cpu_nr=-1 ) {
122
123 int ret;
124 pthread_attr_t attr;
125 struct sched_param schedparam;
126 cpu_set_t cpu_set;
127 size_t dflt_stacksize;
128 _run_loop = 1;
129
130 CPU_ZERO ( &cpu_set );
131 CPU_SET ( cpu_nr,&cpu_set );
132
133 ret = pthread_attr_init(&attr);
134 if (ret != 0) handle_error_en(ret, "pthread_attr_init");
135
136 ret = pthread_attr_setinheritsched ( &attr, PTHREAD_EXPLICIT_SCHED );
137 if (ret != 0) handle_error_en(ret, "pthread_attr_setinheritsched");
138
139 ret = pthread_attr_setschedpolicy ( &attr, schedpolicy );
140 if (ret != 0) handle_error_en(ret, "pthread_attr_setschedpolicy");
141
142 schedparam.sched_priority = priority;
143 ret = pthread_attr_setschedparam ( &attr, &schedparam );
144 if (ret != 0) handle_error_en(ret, "pthread_attr_setschedparam");
145
146 pthread_attr_getstacksize ( &attr, &dflt_stacksize );
147 //DPRINTF ( "default stack size %ld\n", dflt_stacksize );
148 if ( stacksize > 0 ) {
149 ret = pthread_attr_setstacksize ( &attr, stacksize );
150 if (ret != 0) handle_error_en(ret, "pthread_attr_setstacksize");
151
152 }
153
154 ret = pthread_attr_setdetachstate ( &attr, PTHREAD_CREATE_JOINABLE );
155 if (ret != 0) handle_error_en(ret, "pthread_attr_setdetachstate");
156
157 if ( cpu_nr >= 0 ) {
158 ret = pthread_attr_setaffinity_np ( &attr, sizeof ( cpu_set ), &cpu_set );
159 if (ret != 0) handle_error_en(ret, "pthread_attr_setaffinity_np");
160 }
161
162 ret = pthread_create ( &thread_id, &attr, &th_helper, this );
163
164 pthread_attr_destroy ( &attr );
165
166 if ( ret != 0 ) {
167 DPRINTF ( "%s %d %s", __FILE__, __LINE__, name.c_str() );
168 handle_error_en(ret, "pthread_create");
169 }
170 else{
171 joinable=true;
172 }

181inline void * periodic_thread ( EcThread_Ptr th_hook ) {
182
183 int ret = 0;
184 int overruns = 0;
185
186 // thread specific initialization
187 th_hook->th_init ( 0 );
188
189 DPRINTF ( "%s %s period {%ld,%ld} tv\n",
190 __FUNCTION__, th_hook->name.c_str(),
191 th_hook->period.period.tv_sec,
192 th_hook->period.period.tv_usec );
193
194 ret = pthread_setname_np ( pthread_self(), th_hook->name.c_str() );
195 if ( ret != 0 ) {
196 DPRINTF ( "%s : pthread_set_name_np() return code %d\n",
197 th_hook->name.c_str(), ret );
198 exit ( 1 );
199 }
200
201
202#ifdef __COBALT__
203 // PTHREAD_WARNSW, when set, cause the signal SIGXCPU to be sent to the
204 // current thread, whenever it involontary switches to secondary mode;
205 if(th_hook->schedpolicy == SCHED_FIFO){
206 ret = pthread_setmode_np ( 0, PTHREAD_WARNSW, 0 );
207 if ( ret != 0 ) {
208 DPRINTF ( "%s : pthread_set_mode_np() return code %d\n",
209 th_hook->name.c_str(), ret );
210 exit ( 1 );
211 }
212 }
213#endif
214
215 auto start_time = std::chrono::high_resolution_clock::now();
216 auto time = start_time;
217 const auto period = std::chrono::nanoseconds(th_hook->period.period.tv_usec * 1000ULL);
218
219 DPRINTF ( "%s %s : Start looping ...\n",
220 __FUNCTION__, th_hook->name.c_str() );
221
222
223 while ( th_hook->_run_loop ) {
224
225 float time_elapsed_ms = std::chrono::duration<float, std::milli>(time - start_time).count();
226 //DPRINTF("Thread Time elapsed ms: [%f]\n",time_elapsed_ms);
227
228 // thread specific loop
229 th_hook->th_loop ( 0 );
230
231 time = time + period;
232 const auto now = std::chrono::high_resolution_clock::now();
233
234 #if defined(PREEMPT_RT) || defined(__COBALT__)
235 // if less than threshold, print warning (only on rt threads)
236 if(now > time && th_hook->schedpolicy == SCHED_FIFO){
237 ++overruns;
238 DPRINTF("main process overruns: %d\n", overruns);
239 }
240 #endif
241
242 std::this_thread::sleep_until(time);
243 } // end while
244
245 DPRINTF ( "%s %s : exit thread ...\n",
246 __FUNCTION__, th_hook->name.c_str() );
247
248 return 0;
249}
